#[[####################################################################
Targets/CMakeLists.txt

This file is included from the base CMakeLists.txt 
It allows the user to select a hardware target and handles both the 
initialization of the gcc-arm-embedded compiler toolchain and the 
creation of the stm32cubemx linked library, which is autogenerated by 
CubeMX.
#]]####################################################################

# Define a dropdown to allow the user to select a target
set(HARDWARE_TARGET "MCU-h725zgt6" CACHE STRING "Specify the target microprocessor hardware. Each target corresponds to a subdirectory in the Targets directory. See Targets/CMakeLists.txt for implementation.")
set_property(CACHE HARDWARE_TARGET PROPERTY STRINGS "MCU-h725zgt6" "NONE")

if(HARDWARE_TARGET STREQUAL "MCU-h725zgt6")
    message(STATUS "Targeting the MCU-h725zgt6 chip (a.k.a Vaney MainMCU)")
    include("Targets/MCU-h725zgt6/cmake/gcc-arm-none-eabi.cmake")
    add_subdirectory(Targets/MCU-h725zgt6/cmake/stm32cubemx)
elseif(HARDWARE_TARGET STREQUAL "NONE")
    message(WARNING "No hardware target selected! This probably won't compile.")
else() 
    message(SEND_ERROR "Invalid hardware target selected: " ${HARDWARE_TARGET})
endif()

# Add linked libraries
target_link_libraries(${CMAKE_PROJECT_NAME}
    stm32cubemx
)
